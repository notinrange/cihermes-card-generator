// If there are additional recipients, send to them too for (var i = 1; i < toList.length; i++) { GmailApp.sendEmail(toList[i], subject, 'Your email client does not support HTML.', mailOptions); } } // ══════════════════════════════════════════════════════════════════ // SCHEDULE // ══════════════════════════════════════════════════════════════════ function handleSchedule(data) { var sheet = getOrCreateSheet(); var id = Utilities.getUuid(); var scheduleAt = new Date(data.scheduleAt); // ISO string from client var now = new Date(); if (scheduleAt <= now) { throw new Error('Scheduled time is in the past'); } // Store the row // Columns: ID | To | CC | Subject | ImageBase64 | ScheduleAt | Timezone | Status | CreatedAt sheet.appendRow([ id, data.to || '', data.cc || '', data.subject || '', data.imageBase64 || '', scheduleAt.toISOString(), data.timezone || 'UTC', 'pending', now.toISOString() ]); // Ensure the time-driven trigger exists ensureTrigger(); return jsonResponse({ status: 'ok', id: id }); } // ══════════════════════════════════════════════════════════════════ // CANCEL // ══════════════════════════════════════════════════════════════════ function handleCancel(data) { var sheet = getOrCreateSheet(); var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { // skip header row if (rows[i][0] === data.id) { sheet.getRange(i + 1, 8).setValue('cancelled'); return jsonResponse({ status: 'ok', id: data.id }); } } return jsonResponse({ status: 'cancelled', message: 'ID not found (may already be sent)' }); } // ══════════════════════════════════════════════════════════════════ // PROCESS PENDING — called by time trigger every minute // ══════════════════════════════════════════════════════════════════ function processPendingEmails() { var sheet = getOrCreateSheet(); var data = sheet.getDataRange().getValues(); var now = new Date(); var sent = 0; for (var i = 1; i < data.length; i++) { var row = data[i]; var id = row[0]; var to = row[1]; var cc = row[2]; var subject = row[3]; var imageB64 = row[4]; var scheduleAt = new Date(row[5]); var status = row[7]; if (status !== 'pending') continue; if (scheduleAt > now) continue; try { sendCardEmail(to, cc, subject, imageB64); sheet.getRange(i + 1, 8).setValue('sent'); sheet.getRange(i + 1, 9).setValue(now.toISOString()); // sent timestamp sent++; } catch (err) { sheet.getRange(i + 1, 8).setValue('error: ' + err.message); } } if (sent > 0) { Logger.log('processPendingEmails: sent ' + sent + ' email(s)'); } } // ══════════════════════════════════════════════════════════════════ // HELPERS // ══════════════════════════════════════════════════════════════════ function getOrCreateSheet() { var ss = SpreadsheetApp.getActiveSpreadsheet(); sheet.setFrozenRows(1); // Hide the large ImageBase64 column (column E = index 5) sheet.hideColumns(5); // Format sheet.getRange('1:1').setFontWeight('bold'); sheet.setColumnWidth(1, 250); // ID sheet.setColumnWidth(2, 200);
